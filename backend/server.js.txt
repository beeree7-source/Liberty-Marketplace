const express = require("express");
const { registerUser, approveUser } = require("./users");
const app = express();

const PORT = process.env.PORT || 4000;

app.use(express.json());

app.get("/", (req, res) => {
  res.json({ message: "Cigar Order Hub backend is running" });
});

// Register new user (retailer or supplier)
app.post("/api/users/register", registerUser);

// Approve retailer (only suppliers can do this)
app.post("/api/users/:id/approve", approveUser);

app.listen(PORT, () => {
  console.log(`Backend running on port ${PORT}`);
});
